## Quick Start

<PackageManagerTabs>
      <Fragment slot="npm">
      ```shell
      # create a new project with npm
      npm install ascii-progress-bar
      ```
      </Fragment>
      <Fragment slot="pnpm">
      ```shell
      # create a new project with pnpm
      pnpm add ascii-progress-bar
      ```
      </Fragment>
      <Fragment slot="yarn">
      ```shell
      # create a new project with yarn
      yarn add ascii-progress-bar
      ```
      </Fragment>

## Usage Example

```html
<ascii-progress-bar progress="75" pattern="blocks"></ascii-progress-bar>

<script type="module">
  import { AsciiProgressBar } from '@yacosta738/ascii-progress-bar';
  // The component will auto-register
</script>
```
